    \begin{tikzpicture}[shorten >=1pt,node distance=2.5cm,on grid,auto]
      \node[state] (p_0) [] {$p_0$};
      \node[state,accepting] (p_1) [right=of p_0] {$p_1$};
      \node[state,accepting] (p_2) [right=of p_1] {$p_2$};
      \node[state] (q_0) [below=of p_0] {$q_0$};
      \node[state,accepting] (q_1) [right=of q_0] {$q_1$};
      \node[state,accepting] (q_2) [right=of q_1] {$q_2$};
      \path[dashed]
        (p_0) edge node {1} (q_0)
        (p_1) edge node {2} (q_1)
        (p_2) edge node {3} (q_2)
        (p_1) edge node [above left] {4} (q_2); % => p_1 = p_2, q_1 = q_2, p_2 = q_1
      \path[dotted] (p_2) edge node [above right] {5} (q_1); %
      \path[->]
        (q_0) edge node {a,b} (q_1)
        (q_1) edge node {a,b} (q_2)
        (q_2) edge [loop right] node {a,b} (q_2)
        (p_0) edge [bend left=60pt] node {b} (p_2)
        (p_0) edge node {a} (p_1)
        (p_1) edge [bend left] node {a,b} (p_2)
        (p_2) edge [bend left] node [above] {a,b} (p_1);
    \end{tikzpicture}
  \end{figure}
